#Pushing changes to repo
1. git checkout main
2. git pull
3. git checkout -b MODNAME
4. git add .
5. git status
6. git commit -m "DESCRIPTION OF WHAT YOU CHANGED"
7. git push origin BRANCHNAME

#Switching Branches
1. git checkout BRANCHNAME

#Updating Repo to current status
1. git pull origin main
